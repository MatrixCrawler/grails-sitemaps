001package com.bertramlabs.plugins.sitemaps;
002
003import java.util.List;
004
005import org.codehaus.groovy.grails.commons.AbstractGrailsClass;
006import org.codehaus.groovy.grails.commons.GrailsClassUtils;
007
008public class DefaultGrailsSitemapClass extends AbstractGrailsClass implements GrailsSitemapClass {
009
010        public DefaultGrailsSitemapClass(Class<?> clazz) {
011                super(clazz, GrailsSitemapArtefactHandler.SUFFIX);
012        }
013
014        public String getSitemap() {
015                Object sitemapName = GrailsClassUtils.getStaticPropertyValue(getClazz(), "sitemap");
016                if(sitemapName == null) {
017                        return null;
018                }
019                return sitemapName.toString();
020        }
021
022        @SuppressWarnings("rawtypes")
023        public List getSitemapUrls() {
024                return null;
025        }
026
027        public String getDefaultChangeFrequency() {
028                return "monthly";
029        }
030
031        public Double getDefaultPriority() {
032                return 1d;
033        }
034}
